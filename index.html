1. Two sum

Дан список из целых чисел (массив) и число-цель. Найти в списке два числа, сумма которых равна цели, и вывести их порядковые номера (индексы) из первоначального списка (начинается с нуля).

Посмотреть код
var twoSum = function(nums, target) {

 const numsClone = [];

 for (let item in nums) {
  numsClone[item] = nums[item]
 }

 let currNums;

 outer: for (let i=0; i < numsClone.length; i++) {

  for (let x=1; x < numsClone.length; x++) {

   let sum = numsClone[0] + numsClone[x];

   if (sum === target) {
    currNums = [numsClone[0], numsClone[x]];
    break outer;
   }
  };

  numsClone.push(numsClone[0]);
  numsClone.splice(0, 1);
 };

 let indexNums = [nums.indexOf(currNums[0]), nums.lastIndexOf(currNums[1])];

 return indexNums;
}

9. Palindrome Number

Дано число. Нужно проверить, является ли оно палиндромом.

Палиндром — число, которое читается одинаково в любом направлении.

Посмотреть код
var isPalindrome = function(x) {

 let mass = String(x).split('');
 let massReverse = mass.reverse();
 let xReverse = massReverse.join('');

 return x === Number(xReverse);

};
